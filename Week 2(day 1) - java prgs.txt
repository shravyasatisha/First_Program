1. class ProductDiscount {
    public double getFinalAmount(double amount,boolean isMember){
        if(amount>=1000){
            if(isMember){
                double discount = amount * 0.02;
                amount = amount - discount;
            }
            else{
                amount = amount*0.01;
            }
        }
        else{
            System.out.println("No discount !!");
        }
        return amount;
    }
    public static void main(String[] args) {
        ProductDiscount pd = new ProductDiscount();
        System.out.println(pd.getFinalAmount(1000,false));
    }
}

class SmartBillingSlab {
    public double calculateBill(double amount, boolean isFirstOrder){
        if(amount < 500){
            System.out.println("no discount");
            amount=amount;
        }
        else if(amount >= 500 && amount < 1500){
                double discount = amount * 0.05;
                amount = amount - discount;
        }
        else{
            double discount = amount * 0.01;
            amount = amount - discount;
        }
        if(isFirstOrder){
            double discount = amount * 0.02;
            amount = amount - discount ;
        }
        return amount;
    }
    public static void main(String[] args){
        SmartBillingSlab sbs = new SmartBillingSlab();
        System.out.println(sbs.calculateBill(400,true));
    }
}
